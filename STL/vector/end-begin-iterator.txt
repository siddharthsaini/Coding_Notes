#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{

auto it = demo.end(); //returns pointer data-type
cout<<*it;

for (auto i = demo.end()-1; i >= demo.begin(); i--)
{
cout<<*i;
}
return 0;
}
